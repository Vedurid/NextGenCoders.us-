Mini Project: Guess the Number

#include <iostream>
using namespace std;
int main() {
    int number = 7;
    int guess;
    cout << "Guess a number: ";
    cin >> guess;
    if(guess == number) cout << "Correct!";
    else cout << "Try Again!";
    return 0;
}
